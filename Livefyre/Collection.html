def self.create(client, article_id, title, link, tags = nil)
  meta = collectionMeta(client, title, link, tags)
  response = @client.quill.post "/api/v3.0/site/#{client.options[:site_id]}/collection/create/", {:collectionMeta => meta, :articleId => article_id}
  if response.success?
    Conversation.new()
  else
    raise APIException.new(response.body)
  end
end

def update(title, link, tags = nil)
  meta = self.class.collectionMeta(@client, title, link, tags)
  response = @client.quill.post "/api/v3.0/site/#{client.options[:site_id]}/collection/update/", {:collectionMeta => meta, :articleId => @article_id}
  if response.success?
    true
  else
    raise APIException.new(response.body)
  end
end
